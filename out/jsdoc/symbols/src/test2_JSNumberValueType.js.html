1 // This file defines a plain Javascript object called JSNumberValueType that has a single
  2 // number type property called 'value', a getRealValue() method that returns that value,
  3 // and a compareTo() method for comparing two JSNumberValueType instances.
  4 
  5 /**
  6 *@namespace The description of this namespace would go here.
  7 */
  8 
  9 window.multigraph.util.namespace("window.multigraph.valuetype", function (ns) {
 10     "use strict";
 11 
 12 /** @constructor */
 13     var JSNumberValueType = function(value) {
 14         /** a property of the class */
 15         this.value = value;
 16     };
 17     
 18 
 19 /**
 20 * Returns the real value
 21 * @this {number}
 22 * @return {number} The value.
 23 */
 24     JSNumberValueType.prototype.getRealValue = function() {
 25         return this.value;
 26     };
 27 
 28 /**
 29 *Compares the values.
 30 *@method
 31 *@this {number}
 32 *@param {number} x
 33 *@return {number} a value based on comparison
 34 */
 35 
 36     JSNumberValueType.prototype.compareTo = function(x) {
 37         if (this.value < x.value) {
 38             return -1;
 39         } else if (this.value > x.value) {
 40             return 1;
 41         }
 42         return 0;
 43     };
 44     
 45     ns.mixinComparators(JSNumberValueType.prototype);
 46     
 47     ns.JSNumberValueType = JSNumberValueType;
 48 
 49 });
